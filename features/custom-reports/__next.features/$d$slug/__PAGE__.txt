        function triggerInteractiveRefresh() {
            if (isRefreshing) return;

            const progressWrapper = document.getElementById('sim-progress-wrapper');

            // Simulate progress
            let progress = 0;
            const interval = setInterval(() => {
                progress += Math.floor(Math.random() * 15) + 5;
                if (progress >= 100) {
                    progress = 100;
                    clearInterval(interval);

                    // Stop refresh UI
                    setTimeout(() => {
                        icon.classList.remove('animate-spin');
                        progressWrapper.classList.add('hidden');
                        isRefreshing = false;

                        // Update values with new refreshed numbers
                        updateSimulationNumbers();

                        // Flash cells
                        document.querySelectorAll('[id^="sim-col-"]').forEach(cell => {
                            cell.classList.add('flash-update');
                            setTimeout(() => cell.classList.remove('flash-update'), 1000);
                        });

                        // Get current time
                        const now = new Date();
                        const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' });
                        timestampText.innerHTML = `Sync Status: <span class="text-emerald-600 font-bold">Refreshed at ${timeStr}</span>`;
                    }, 400);
                }
                progressBar.style.width = `${progress}%`;
                progressPct.innerText = `${progress}%`;
            }, 100);
        }
