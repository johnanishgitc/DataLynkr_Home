            const setupSynchronizedPlayback = (videoMobile, videoLaptop) => {

                    videoLaptop.currentTime = 0;

                    const playMobile = videoMobile.play();
                    const playLaptop = videoLaptop.play();

                    Promise.all([playMobile, playLaptop]).catch(error => {
                        console.log('Synchronized video play was interrupted or blocked:', error);
                    });
                };

                const checkReadyStates = () => {
                    if (videoMobile.readyState >= 3) mobileReady = true;
                    if (videoLaptop.readyState >= 3) laptopReady = true;

                    if (mobileReady && laptopReady) {
                        startPlayback();
                    }
                };

                videoMobile.addEventListener('canplay', () => {
                    mobileReady = true;
                    if (laptopReady) startPlayback();
                });

                videoLaptop.addEventListener('canplay', () => {
                    laptopReady = true;
                    if (mobileReady) startPlayback();
                });

                // Synchronized looping logic: when mobile video (primary driver) ends, restart both at the same time
                videoMobile.addEventListener('ended', () => {
                    videoMobile.currentTime = 0;
                    videoLaptop.currentTime = 0;
                    const playMobile = videoMobile.play();
                    const playLaptop = videoLaptop.play();
                    Promise.all([playMobile, playLaptop]).catch(error => {
                        console.log('Synchronized video loop play was interrupted or blocked:', error);
                    });
                });

                checkReadyStates();

                // Safety timeout in case load events don't fire
                setTimeout(() => {
                    if (!playAttempted) {
                        console.log('Safety timeout: starting synchronized video playback');
                        startPlayback();
                    }
                }, 3000);
